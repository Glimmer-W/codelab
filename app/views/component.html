                        angular.module('yn.ui.buttons')
                        .directive('ynUiBtnMoreCon', function () {
                          return {
                              restrict: "AE",
                              scope: {
                                  isMore: "=?" //是否展示更多条件
                              },
                              // 设置模板
                              templateUrl: 'yn/templates/directives/buttons/morecondition/morecondition.html'
                          };
                        });

参数

参数 类型 详情
isMore boolean 可绑定数据的angular表达式，控制是否显示“更多”条件，默认不显示
